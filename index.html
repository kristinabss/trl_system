class TRLClassifier: """Классификатор уровня готовности технологий""" def preprocess_data(self, product_data: Dict) -> np.ndarray: # Расчёт коэффициента вариации технических параметров tech_params = product_data.get('technical_characteristics', {}) cv = np.std(values) / np.mean(values) * 100 # Расчёт индекса технического совершенства (ИТС) ratios = [tech_params[k] / target_params[k] for k in tech_params] technical_index = np.mean(ratios) return np.array([cv, cumulative_volume, technical_index, cost_reduction, first_pass_yield, annual_volume, profitability]).reshape(1, -1) def predict(self, product_data: Dict) -> Dict: features = self.preprocess_data(product_data) trl_prediction = self.model.predict(features)[0] trl_probabilities = self.model.predict_proba(features)[0] confidence = np.sort(trl_probabilities)[::-1][0] - np.sort(trl_probabilities)[::-1][1] return {'trl_level': int(trl_prediction), 'confidence': float(confidence), 'interpretation': self._interpret_trl(int(trl_prediction)), 'recommendations': self._generate_recommendations(...)}

Методология УГТ

Девятиуровневая шкала оценки технологической готовности · Минпромторг РФ · NASA

Методология УГТ разработана NASA в 1970-х гг. для оценки зрелости технологий — от фундаментальных исследований до серийного производства. В РФ закреплена методическими рекомендациями Минпромторга РФ (2023) и адаптирована для гражданских отраслей промышленности.
УГТ 1
Базовые принципы
Теоретическое обоснование возможности технического решения, фундаментальные исследования.
УГТ 2
Концепция технологии
Формирование концепции и определение областей потенциального применения.
УГТ 3
Экспериментальное подтверждение
Экспериментальное подтверждение концепции в лабораторных условиях.
УГТ 4
Лабораторная валидация
Валидация в лаборатории: функционирование отдельных компонентов или подсистем.
УГТ 5
Приближённая среда
Валидация технологии в условиях, максимально приближённых к реальной среде эксплуатации.
УГТ 6
Релевантная среда
Технология демонстрируется в релевантной среде. Опытная партия изготовлена.
УГТ 7
Операционная среда
Технология в операционной среде, приближённой к условиям промышленности. Мелкая серия.
УГТ 8
Квалификация
Технология проходит квалификацию посредством испытаний в реальных условиях применения.
УГТ 9
Серийное производство
Технология доказала работоспособность и готова к крупносерийному коммерческому производству.
